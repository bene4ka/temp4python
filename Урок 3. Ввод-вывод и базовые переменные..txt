#Задание 2
stages = []
stage_names = [
    "первый", "второй", "третий", "четвёртый", 
    "пятый", "шестой", "седьмой (последний)"
]

for name in stage_names:
    stage = input(f"Введите {name} этап развития человека: ")
    stages.append(stage)

print(*stages, sep='+')
