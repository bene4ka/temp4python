#Задание 1
number = int(input("Введите целое число: "))

if number == 0:
    print("нулевое число")
else:
    if number > 0:
        sign = "положительное"
    else:
        sign = "отрицательное"

    if number % 2 == 0:
        print(f"{sign} четное число")
    else:
        print("число не является четным")

#Задание 2
word = input("Введите слово из маленьких латинских букв: ")

vowels = ['a', 'e', 'i', 'o', 'u']

vowel_count = sum(1 for letter in word if letter in vowels)
consonant_count = len(word) - vowel_count

print(f"Гласных: {vowel_count}")
print(f"Согласных: {consonant_count}")

for vowel in vowels:
    count = word.count(vowel)
    print(f"'{vowel}': {count if count > 0 else False}")
